-- 3. Creación de la tabla Planes (Catálogo)
CREATE TABLE Planes (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre NVARCHAR(50) NOT NULL,
    Precio DECIMAL(10,2) NOT NULL,
    DuracionMeses INT NOT NULL
);
GO

-- 4. Creación de la tabla Suscripciones (Transaccional)
CREATE TABLE Suscripciones (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    ClienteNombre NVARCHAR(100) NOT NULL,
    DocumentoIdentidad NVARCHAR(15) NOT NULL,
    PlanId INT NOT NULL,
    FechaInicio DATETIME NOT NULL,
    FechaFin DATETIME NOT NULL,
    Estado NVARCHAR(20) NOT NULL,
    FOREIGN KEY (PlanId) REFERENCES Planes(Id)
);
GO

-- 5. Creación de la tabla Usuarios (Seguridad / Login)
CREATE TABLE Usuarios (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    NombreCompleto NVARCHAR(100) NOT NULL,
    Correo NVARCHAR(50) UNIQUE NOT NULL,
    Clave NVARCHAR(50) NOT NULL, -- (Nota: En producción se recomienda encriptar)
    Rol NVARCHAR(20) NOT NULL
);
GO

-- 8. Procedimiento: Registrar Suscripción
CREATE PROCEDURE sp_RegistrarSuscripcion
    @ClienteNombre NVARCHAR(100),
    @DocumentoIdentidad NVARCHAR(15),
    @PlanId INT,
    @FechaInicio DATETIME,
    @FechaFin DATETIME,
    @Estado NVARCHAR(20)
AS
BEGIN
    INSERT INTO Suscripciones (ClienteNombre, DocumentoIdentidad, PlanId, FechaInicio, FechaFin, Estado) 
    VALUES (@ClienteNombre, @DocumentoIdentidad, @PlanId, @FechaInicio, @FechaFin, @Estado);
END
GO

-- 9. Procedimiento: Listar Suscripciones
CREATE PROCEDURE sp_ListarSuscripciones
AS
BEGIN
    SELECT 
        s.Id, 
        s.ClienteNombre, 
        s.DocumentoIdentidad, 
        s.PlanId, 
        s.FechaInicio, 
        s.FechaFin, 
        s.Estado, 
        p.Nombre AS NombrePlan
    FROM Suscripciones s
    INNER JOIN Planes p ON s.PlanId = p.Id;
END
GO

-- 10. Procedimiento: Obtener Duración de un Plan
CREATE PROCEDURE sp_ObtenerDuracionPlan
    @Id INT
AS
BEGIN
    SELECT DuracionMeses 
    FROM Planes 
    WHERE Id = @Id;
END
GO

-- 11. Procedimiento: Validar Usuario para Login
CREATE PROCEDURE sp_ValidarUsuario
    @Correo NVARCHAR(50),
    @Clave NVARCHAR(50)
AS
BEGIN
    SELECT Id, NombreCompleto, Correo, Rol 
    FROM Usuarios 
    WHERE Correo = @Correo AND Clave = @Clave;
END
GO

    @Id INT,
    @Nombre NVARCHAR(50),
    @Precio DECIMAL(10,2),
    @DuracionMeses INT

        DuracionMeses = @DuracionMeses

-- SP: Eliminar un plan (solo si no tiene suscripciones asociadas)
CREATE PROCEDURE sp_EliminarPlan
    @Id INT
AS
BEGIN
    IF NOT EXISTS (SELECT 1 FROM Suscripciones WHERE PlanId = @Id)
    BEGIN
        DELETE FROM Planes WHERE Id = @Id;
    END
    ELSE
    BEGIN
        RAISERROR('No se puede eliminar el plan porque tiene suscripciones asociadas.', 16, 1);
    END
END
GO

-- Tabla Pagos (Registro de cobros asociados a suscripciones)
CREATE TABLE Pagos (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    SuscripcionId INT NOT NULL,
    Monto DECIMAL(10,2) NOT NULL,
    FechaPago DATETIME NOT NULL DEFAULT GETDATE(),
    MetodoPago NVARCHAR(30) NOT NULL,
    Referencia NVARCHAR(50) NULL,
    Observaciones NVARCHAR(200) NULL,
    FOREIGN KEY (SuscripcionId) REFERENCES Suscripciones(Id)
);
GO

-- SP: Registrar un pago
CREATE PROCEDURE sp_RegistrarPago
    @SuscripcionId INT,
    @Monto DECIMAL(10,2),
    @FechaPago DATETIME,
    @MetodoPago NVARCHAR(30),
    @Referencia NVARCHAR(50),
    @Observaciones NVARCHAR(200)
AS
BEGIN
    INSERT INTO Pagos (SuscripcionId, Monto, FechaPago, MetodoPago, Referencia, Observaciones)
    VALUES (@SuscripcionId, @Monto, @FechaPago, @MetodoPago, @Referencia, @Observaciones);
END
GO

-- SP: Listar todos los pagos con datos de la suscripción
CREATE PROCEDURE sp_ListarPagos
AS
BEGIN
    SELECT 
        pg.Id,
        pg.SuscripcionId,
        pg.Monto,
        pg.FechaPago,
        pg.MetodoPago,
        pg.Referencia,
        pg.Observaciones,
        s.ClienteNombre,
        p.Nombre AS NombrePlan
    FROM Pagos pg
    INNER JOIN Suscripciones s ON pg.SuscripcionId = s.Id
    INNER JOIN Planes p ON s.PlanId = p.Id
    ORDER BY pg.FechaPago DESC;
END
GO

-- SP: Listar pagos de una suscripción específica
CREATE PROCEDURE sp_ListarPagosPorSuscripcion
    @SuscripcionId INT
AS
BEGIN
    SELECT 
        pg.Id,
        pg.SuscripcionId,
        pg.Monto,
        pg.FechaPago,
        pg.MetodoPago,
        pg.Referencia,
        pg.Observaciones,
        s.ClienteNombre,
        p.Nombre AS NombrePlan
    FROM Pagos pg
    INNER JOIN Suscripciones s ON pg.SuscripcionId = s.Id
    INNER JOIN Planes p ON s.PlanId = p.Id
    WHERE pg.SuscripcionId = @SuscripcionId
    ORDER BY pg.FechaPago DESC;
END
GO

-- SP: Obtener un pago por su Id

-- ==========================================
-- SECCIÓN 7: DASHBOARD (Consultas de Resumen)
-- ==========================================

-- SP: Obtener los indicadores principales del dashboard
CREATE PROCEDURE sp_ObtenerResumenDashboard
AS
BEGIN
    SELECT 
        (SELECT COUNT(*) FROM Clientes WHERE Activo = 1) AS TotalClientes,
        (SELECT COUNT(*) FROM Suscripciones WHERE Estado = 'Activo') AS SuscripcionesActivas,
        (SELECT COUNT(*) FROM Suscripciones WHERE Estado = 'Cancelado') AS SuscripcionesCanceladas,
        (SELECT ISNULL(SUM(Monto), 0) FROM Pagos WHERE MONTH(FechaPago) = MONTH(GETDATE()) AND YEAR(FechaPago) = YEAR(GETDATE())) AS IngresosMensuales;
END
GO

-- SP: Obtener suscripciones que vencen en los próximos 30 días
CREATE PROCEDURE sp_ObtenerSuscripcionesPorVencer
AS
BEGIN
    SELECT 
        s.Id,
        s.ClienteNombre,
        s.DocumentoIdentidad,
        s.PlanId,
        s.FechaInicio,
        s.FechaFin,
        s.Estado,
        p.Nombre AS NombrePlan
    FROM Suscripciones s
    INNER JOIN Planes p ON s.PlanId = p.Id
    WHERE s.Estado = 'Activo' 
      AND s.FechaFin BETWEEN GETDATE() AND DATEADD(DAY, 30, GETDATE())
    ORDER BY s.FechaFin ASC;
END
GO

-- SP: Obtener distribución de suscripciones activas por plan
CREATE PROCEDURE sp_ObtenerDistribucionPorPlan
AS
BEGIN
    SELECT 
        p.Nombre AS NombrePlan,
        COUNT(s.Id) AS CantidadSuscripciones
    FROM Planes p
    LEFT JOIN Suscripciones s ON p.Id = s.PlanId AND s.Estado = 'Activo'
    GROUP BY p.Nombre
    ORDER BY CantidadSuscripciones DESC;
END
GO

-- ==========================================

-- Tabla Bitacora (Registro de acciones del sistema)
CREATE TABLE Bitacora (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    UsuarioId INT NULL,
    NombreUsuario NVARCHAR(100) NOT NULL,
    Accion NVARCHAR(100) NOT NULL,
    Detalle NVARCHAR(300) NULL,
    FechaHora DATETIME NOT NULL DEFAULT GETDATE(),
    FOREIGN KEY (UsuarioId) REFERENCES Usuarios(Id)
);
GO

-- SP: Registrar una entrada en la bitácora
CREATE PROCEDURE sp_RegistrarBitacora
    @NombreUsuario NVARCHAR(100),
    @Accion NVARCHAR(100),
    @Detalle NVARCHAR(300)
AS
BEGIN
    INSERT INTO Bitacora (NombreUsuario, Accion, Detalle)
    VALUES (@NombreUsuario, @Accion, @Detalle);
END
GO

-- SP: Listar entradas de la bitácora con filtro opcional por fechas
CREATE PROCEDURE sp_ListarBitacora
    @FechaDesde DATETIME = NULL,
    @FechaHasta DATETIME = NULL
AS
BEGIN
    SELECT Id, NombreUsuario, Accion, Detalle, FechaHora
    FROM Bitacora
    WHERE (@FechaDesde IS NULL OR FechaHora >= @FechaDesde)
      AND (@FechaHasta IS NULL OR FechaHora <= @FechaHasta)
    ORDER BY FechaHora DESC;
END
