-- 3. Creación de la tabla Planes (Catálogo)
CREATE TABLE Planes (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre NVARCHAR(50) NOT NULL,
    Precio DECIMAL(10,2) NOT NULL,
    DuracionMeses INT NOT NULL
);
GO

-- 4. Creación de la tabla Suscripciones (Transaccional)
CREATE TABLE Suscripciones (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    ClienteNombre NVARCHAR(100) NOT NULL,
    DocumentoIdentidad NVARCHAR(15) NOT NULL,
    PlanId INT NOT NULL,
    FechaInicio DATETIME NOT NULL,
    FechaFin DATETIME NOT NULL,
    Estado NVARCHAR(20) NOT NULL,
    FOREIGN KEY (PlanId) REFERENCES Planes(Id)
);
GO

-- 5. Creación de la tabla Usuarios (Seguridad / Login)
CREATE TABLE Usuarios (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    NombreCompleto NVARCHAR(100) NOT NULL,
    Correo NVARCHAR(50) UNIQUE NOT NULL,
    Clave NVARCHAR(50) NOT NULL, -- (Nota: En producción se recomienda encriptar)
    Rol NVARCHAR(20) NOT NULL
);
GO

-- 8. Procedimiento: Registrar Suscripción
CREATE PROCEDURE sp_RegistrarSuscripcion
    @ClienteNombre NVARCHAR(100),
    @DocumentoIdentidad NVARCHAR(15),
    @PlanId INT,
    @FechaInicio DATETIME,
    @FechaFin DATETIME,
    @Estado NVARCHAR(20)
AS
BEGIN
    INSERT INTO Suscripciones (ClienteNombre, DocumentoIdentidad, PlanId, FechaInicio, FechaFin, Estado) 
    VALUES (@ClienteNombre, @DocumentoIdentidad, @PlanId, @FechaInicio, @FechaFin, @Estado);
END
GO

-- 9. Procedimiento: Listar Suscripciones
CREATE PROCEDURE sp_ListarSuscripciones
AS
BEGIN
    SELECT 
        s.Id, 
        s.ClienteNombre, 
        s.DocumentoIdentidad, 
        s.PlanId, 
        s.FechaInicio, 
        s.FechaFin, 
        s.Estado, 
        p.Nombre AS NombrePlan
    FROM Suscripciones s
    INNER JOIN Planes p ON s.PlanId = p.Id;
END
GO

-- 10. Procedimiento: Obtener Duración de un Plan
CREATE PROCEDURE sp_ObtenerDuracionPlan
    @Id INT
AS
BEGIN
    SELECT DuracionMeses 
    FROM Planes 
    WHERE Id = @Id;
END
GO

-- 11. Procedimiento: Validar Usuario para Login
CREATE PROCEDURE sp_ValidarUsuario
    @Correo NVARCHAR(50),
    @Clave NVARCHAR(50)
AS
BEGIN
    SELECT Id, NombreCompleto, Correo, Rol 
    FROM Usuarios 
    WHERE Correo = @Correo AND Clave = @Clave;
END
GO

-- 1. Procedimiento para obtener una suscripción específica
CREATE PROCEDURE sp_ObtenerSuscripcionPorId
    @Id INT
AS
BEGIN
    SELECT 
        s.Id, 
        s.ClienteNombre, 
        s.DocumentoIdentidad, 
        s.PlanId, 
        s.FechaInicio, 
        s.FechaFin, 
        s.Estado, 
        p.Nombre AS NombrePlan
    FROM Suscripciones s
    INNER JOIN Planes p ON s.PlanId = p.Id
    WHERE s.Id = @Id;
END
GO

-- 2. Procedimiento para Anular (Soft Delete)
CREATE PROCEDURE sp_AnularSuscripcion
    @Id INT
AS
BEGIN
    UPDATE Suscripciones
    SET Estado = 'Cancelado'
    WHERE Id = @Id;
END
GO

-- 3. Procedimiento para Renovar (Update)
CREATE PROCEDURE sp_RenovarSuscripcion
    @Id INT,
    @NuevoPlanId INT,
    @NuevaFechaFin DATETIME
AS
BEGIN
    UPDATE Suscripciones
    SET 
        PlanId = @NuevoPlanId,
        FechaFin = @NuevaFechaFin,
        Estado = 'Activo'
    WHERE Id = @Id;
END
GO

-- 4. Procedimiento para Registrar Usuario
CREATE PROCEDURE sp_RegistrarUsuario
    @NombreCompleto NVARCHAR(100),
    @Correo NVARCHAR(50),
    @Clave NVARCHAR(50),
    @Rol NVARCHAR(20)
AS
BEGIN
    INSERT INTO Usuarios (NombreCompleto, Correo, Clave, Rol)
    VALUES (@NombreCompleto, @Correo, @Clave, @Rol);
    
    SELECT SCOPE_IDENTITY();
END
GO
